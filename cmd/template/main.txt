package main

import (
    "fmt"
	"github.com/danvolchek/AdventOfCode/lib"
	"os"
	"path"
)

func input() *os.File {
	input, err := os.Open(path.Join("{{.Year}}", "{{.Day}}", "input.txt"))
	if err != nil {
		panic(err)
	}

	return input
}

func parse(line string) any {
    return line
}

func solve(lines []any) any {
	for _, line := range lines {

        fmt.Println(line)
    }

    return 0
}

func main() {
	lib.TestSolveParseLines("foo", parse, solve)
    lib.SolveParseLines(input(), parse, solve)
}
